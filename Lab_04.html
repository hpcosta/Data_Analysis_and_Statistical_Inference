# create a clean version of the 'gained' data, dropping NAs
gained_clean = na.omit(nc$gained)
# set the variable, n, to the length of the clean gained vector
n = length(gained_clean)
# bootstrap the clean sample 1) initialize boot_means object with 100 NAs
boot_means = rep(NA, 100)
# 2) create loop for sampling with replacement of size 'n' for 100
# iterations
for (i in 1:100) {
    samp = sample(gained_clean, n, replace = TRUE)
    boot_means[i] = mean(samp)
}
# create histogram of 'boot_means'
hist(boot_means)

# use inference function to determine whether there is a statistical
# difference in the mean birth weight between the smokers and the
# non-smokers y is response/variable of interest, x is the
# categorical/explanatory variable est is the parameter of interest:
# 'mean' in this case (but could also be 'median' or 'proportion') type is
# the type of inference: 'ht' for hypothesis test or 'ci' for confidence
# interval set the null value; 0 in this case as the H0 is that the
# populations are equal alternative could be 'twosided', 'less', 'greater'
# method of inference could be 'simulation' or 'theoretical'
inference(y = nc$weight, x = nc$habit, est = "mean", type = "ht", null = 0, 
    alternative = "twosided", method = "theoretical")

## Response variable: numerical, Explanatory variable: categorical
## Difference between two means
## Summary statistics:
## n_nonsmoker = 873, mean_nonsmoker = 7.144, sd_nonsmoker = 1.519

## Observed difference between means (nonsmoker-smoker) = 0.3155
## H0: mu_nonsmoker - mu_smoker = 0 
## HA: mu_nonsmoker - mu_smoker != 0 
## Standard error = 0.134 
## Test statistic: Z =  2.359 
## p-value =  0.0184
