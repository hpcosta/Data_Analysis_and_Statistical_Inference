# make a plot of the ME as a function of p 1) make a vector p that is a
# sequence from 0 to 1 with each number separated by 0.01:
n = 1000
p = seq(0, 1, 0.01)
# 2) create a vector of the margin of error (me) associated with each of
# these values of p using the familiar approximate formula (ME = 2 X SE)
me = 2 * sqrt(p * (1 - p)/n)
# 3) plot the two vectors against each other to reveal their relationship
plot(me ~ p)
